local function round(n: number, decimals: number): number
	local m = 10 ^ (decimals or 0)
	return math.floor(n * m + 0.5) / m
end

local function genStats(tier: number, rank: number): GenStats
	local r = RANKS[rank]
	local a = ARCH[r.archetype]
	local step = tier - 1
	local rp = rankPower(rank)

	local grabSpeed     = (BASE.grabSpeedTop - BASE.grabSpeedDrop * (step / 11)) * a.grabSpeed / rp
	local weightCap     = BASE.weightCapStart * (BASE.weightCapGrowth ^ step) * a.weightCap * rp
	local luck          = (BASE.luckPerTier * step) * a.luck * rp
	local qualityBias   = (BASE.qualityPerTier * step) * a.qualityBias * rp
	local modifierChance= (BASE.modifierPerTier * step) * a.modifierChance * rp
	local stability     = (BASE.stabilityStart + BASE.stabilityPerTier * step) * a.stability * rp
	local multiGrab     = (BASE.multiGrabPerTier * step) * a.multiGrab * rp

	-- specialEfficiency : { rarity, mult } (modele existant, pilote par RANKS[rank].rarity).
	-- "all" = universel ; une rarete de scrap = bonus de valeur cible ; nil = aucun.
	local special: SpecialEfficiency? = nil
	if r.rarity == "all" then
		special = { rarity = "all", mult = 1.0 + 0.02 * tier }
	elseif r.rarity ~= nil then
		special = { rarity = r.rarity, mult = 1.0 + 0.15 + 0.04 * tier }
	end

	return {
		grabSpeed = round(math.max(0.35, grabSpeed), 2),
		luck = round(luck, 3),
		qualityBias = round(qualityBias, 3),
		modifierChance = round(modifierChance, 3),
		weightCap = round(weightCap, 1),
		stability = round(math.clamp(stability, 0, 0.98), 3),
		multiGrab = round(math.min(multiGrab, 0.95), 3),
		specialEfficiency = special,
	}
end

-- Note global (0..100), STRICTEMENT croissante en (tier, puis rank) -> garantit "du moins bon
-- au meilleur" pour le tri/affichage de l'album. Les stats individuelles, elles, ont de la saveur.
local function ratingOf(tier: number, rank: number): number
	local idx = (tier - 1) * 10 + (rank - 1) -- 0..119
	return round(idx / 119 * 100, 1)
end

local function rarityWeight(tier: number, luck: number?): number
	local w = RARITY_WEIGHT_BASE / (RARITY_WEIGHT_FALLOFF ^ (tier - 1))
	if luck and luck > 0 then
		w *= (1 + luck * (tier - 1) * 0.08)
	end
	return w
end

end

		end
	end
	return nil
end
